//this example is from chapter 3 on recursion //this function determines whether a number is even using recursion //this is the code I wrote function isEven(number) { if (number == 0) { return true; } else if (number == 1) { return false; } else { return isEven(number - 2); } } //the solution from the book gives the solution for the input -1 //it makes sense because if 'n' is below zero, then returning '-n' cancels the negative out and returns it as positive //without that line, the stack runs forever and crashes function isEven(n) { if (n == 0) return true; else if (n == 1) return false; else if (n < 0) return isEven(-n); else return isEven(n - 2); }
